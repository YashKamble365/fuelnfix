Backend Server Entry Point (index.js)

Component Overview

This is the heart of the backend. It initializes the Express app, connects to MongoDB, and most importantly, sets up the Socket.IO server for real-time communication. Queries from clients for live tracking and status updates pass through here.

Critical Logic: The provider_location_update event listener is what makes the "Uber-like" tracking possible. It broadcasts location to the user and updates MongoDB for geospatial queries simultaneously. 
// server/index.js - Socket.IO & Server Setup
const express = require('express');
const { Server } = require("socket.io");
const User = require('./models/User');

// Initialize Socket.IO with CORS
const io = new Server(server, {
    cors: { origin: "*" }
});

// --- REAL-TIME EVENT HANDLING ---
io.on('connection', (socket) => {
    
    // 1. Join Specific Rooms (e.g., Request ID)
    socket.on('join_room', (roomId) => {
        socket.join(roomId);
    });

    // 2. LIVE TRACKING LOGIC
    socket.on('provider_location_update', async ({ roomId, userId, location }) => {
        // A. Broadcast to specific request room (User sees this)
        if (roomId) {
            io.to(roomId).emit('track_provider', location);
        }

        // B. Update Database for "Search Nearby" functionality
        if (userId) {
            await User.findByIdAndUpdate(userId, {
                liveLocation: {
                    type: 'Point',
                    coordinates: [location.lng, location.lat]
                },
                isOnline: true
            });
        }
    });

    // 3. Status Updates (Accepted -> Arrived -> Completed)
    socket.on('update_status', ({ roomId, status }) => {
        io.to(roomId).emit('status_changed', status);
    });
});

Backend The Matching Algorithm (requestController.js)

Component Overview

This controller handles the logic for finding providers. It uses MongoDB Geospatial Queries ($near) to find providers within a radius, sorts them by distance, and calculates dynamic pricing based on the distance.

// server/controllers/requestController.js - searchProviders function
exports.searchProviders = async (req, res) => {
    const { serviceName, originLat, originLng, category } = req.body;

    // 1. Geospatial Query
    const query = {
        role: 'provider',
        isVerified: true,
        isOnline: true,
        providerCategory: category, // e.g., 'Mechanic'
        location: {
            $near: {
                $geometry: { type: "Point", coordinates: [originLng, originLat] },
                $maxDistance: 50000 // 50km Radius
            }
        },
        // Check if provider offers the specific service (e.g., 'Flat Tire')
        services: {
            $elemMatch: { name: serviceName, active: true }
        }
    };

    const providers = await User.find(query).limit(50);

    // 2. Dynamic Pricing Calculation
    const results = providers.map(provider => {
        // Haversine Formula for precise distance
        const distance = calculateHaversineDistance(
            originLat, originLng, 
            provider.location.coordinates[1], 
            provider.location.coordinates[0]
        );

        // Get rates (simplified for demo)
        const baseFee = 300; 
        const pricePerKm = 10;
        const totalEstimate = baseFee + (distance * pricePerKm);

        return {
            provider,
            distance: distance.toFixed(2),
            totalEstimate: Math.ceil(totalEstimate)
        };
    });

    res.json(results);
};

Frontend Provider Dashboard (ProviderDashboard.jsx)

Component Overview

The control center for mechanics. It handles incoming request popups (via Socket.IO), job acceptance, and the "Ghost Driver" simulation logic which animates the car movement along a real road path (using OSRM).

// client/src/pages/ProviderDashboard.jsx - Simulation Logic
const startSimulation = async () => {
    // 1. Fetch Route from OSRM (Open Source Routing Machine)
    const osrmUrl = `https://router.project-osrm.org/route/v1/driving/${start};${end}`;
    const res = await fetch(osrmUrl);
    const json = await res.json();
    const pathPoints = decodePolyline(json.routes[0].geometry);

    // 2. Animation Loop
    const animate = () => {
        // Calculate next position based on speed (km/h)
        const nextPoint = calculateNextPoint(pathPoints, speed);
        
        // Update Local State (Provider sees car moving)
        setUser(prev => ({ ...prev, liveLocation: nextPoint }));

        // Broadcast to Server (User sees car moving)
        socket.emit('provider_location_update', {
            roomId: activeJob._id,
            location: nextPoint
        });

        requestAnimationFrame(animate);
    };
    requestAnimationFrame(animate);
};
